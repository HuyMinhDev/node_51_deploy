## CONTAINER
# show tất cả container
    docker container list
    docker container ls
     -a : show tất cả các container kể cả container đang ofline
        docker container list -a
        docker container ls -a
        docker ps

# tạo/chạy container
    docker container run --env-file .env --name con-be_cyber_community -p 3070:3069 -d img-be_cyber_community
    --name: đặt tên cho container
    -p: quy định cổng mà fe sẽ kết nối 
    -d: chạy ngầm, ko chiếm dụng terminal
    TÊN IMAGE LUÔN ĐỂ CUỐI CÙNG
    kiểm tra xem image đã có trong máy hay chưa
        _ có: Ưu tiên dùng trong máy trước
        _ chưa: tự lên docker hub tìm và tải về
    --env-file: đọc dử liệu bên trong file env

# Kiểm tra log của container (source)
    docker logs id_ten_container
    docker logs -f id_ten_container
    -f: bắt dính cập nhật log mới nhất (chỉ khi container start)

# Xóa container
    docker container remove id_ten_container
